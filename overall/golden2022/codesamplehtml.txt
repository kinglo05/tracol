<script>
    function toggleSidebar() {
        var sidebar = document.getElementById("sidebar");
        var mainContent = document.getElementById("mainContent");
        if (sidebar.style.left === "-250px" || sidebar.style.left === "") {
            sidebar.style.left = "0";
            mainContent.style.marginLeft = "250px";
        } else {
            sidebar.style.left = "-250px";
            mainContent.style.marginLeft = "0";
        }
    }

    

    function generateReport() {
    const table1 = document.getElementById("payments-table").outerHTML;
    const table2 = document.getElementById("payments-table2").outerHTML;

    const totalCollected = document.getElementById("totalCollected").innerText;
    const totalExpenses = document.getElementById("totalExpens").innerText;
    const finalNabilin = document.getElementById("finalNabilin").innerText;

    const reportWindow = window.open("", "_blank", "width=800,height=1000");

    const htmlContent = `
        <html>
        <head>
            <title>Golden Transactions Report</title>
            <style>
                body { font-family: Arial, sans-serif; padding: 20px; }
                h2 { color: #2e2929; }
                table { width: 100%; border-collapse: collapse; margin-bottom: 20px; }
                th, td { border: 1px solid #333; padding: 8px; text-align: left; font-size: 14px; }
                th { background-color: #f0f0f0; }
                .totals { margin-top: 20px; }
                .totals p { font-size: 16px; margin: 5px 0; }
            </style>
        </head>
        <body>
            <h2>Golden Transactions Report</h2>
            <h3>Collections</h3>
            ${table1}
            <h3>Expenses</h3>
            ${table2}
            <div class="totals">
                <p><strong>Total Collection:</strong> ${totalCollected}</p>
                <p><strong>Total Expenses:</strong> ${totalExpenses}</p>
                <p><strong>Nabilin/Kulang:</strong> ${finalNabilin}</p>
            </div>
        </body>
        </html>
    `;

    reportWindow.document.open();
    reportWindow.document.write(htmlContent);
    reportWindow.document.close();

    // Wait until content loads, then print
    reportWindow.onload = function () {
        reportWindow.print();
        reportWindow.onafterprint = function () {
            reportWindow.close();
        };
    };
}
